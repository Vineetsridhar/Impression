102def create_new_user(data): 

103 try: 

104 fields = ["email", "given_name", "family_name", "picture"] 

105 for field in fields: 

106 if field not in data: 

107 data[field] = "" 

108 imp_util.users.new_user( 

109 data["email"], data["given_name"], data["family_name"], data["picture"] 

110 ) 

111 imp_util.qr.create_new_qr_code(data["email"]) 

112 return {"success": True, "email": data["email"]} 

113 except Exception as err: 

114 print(err) 

115 return {"success": False}
